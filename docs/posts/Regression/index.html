Survival Analysis

#as 'Attrition' is a categorical variable, the following command changes it to a binary 1/0

employee.att$event <- with(employee.att,ifelse(Attrition=="Yes", 1, 0))

time<-employee.att$YearsAtCompany #define time

event<-employee.att$event #define event

group<-employee.att$OverTime #define groups to compare multiple survival curves

survival<-Surv(time,event) #creates the survival object
